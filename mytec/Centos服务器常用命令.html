# Mysql 忘记密码重置

  1. 停止Mysql
sudo systemctl stop mysqld
  1. 编辑 Mysql配置文件
vi /etc/my.cnf
  1. 添加
skip-grant-tables
  1. 启动 Mysql
sudo systemctl start mysqld
  1. 修改密码
update user set authentication_string= PASSWORD("Orchid123!") where User='root';
flush privileges;
exit;
  1. 重启 Mysql
sudo systemctl restart mysqld
